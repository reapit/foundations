## Usage

### List with custom renderer item

<Playground>
  {() => {
    const items = [{ value: 'Viewing' }, { value: '11:00 AM - 12:00 PM' }, { value: 'Info about the viewing' }]
    const renderItem = item => {
      return <div>{item.value}</div>
    }
    return (
      <DocsWrapper>
        <List.IconList items={items} renderItem={renderItem} />
      </DocsWrapper>
    )
  }}
</Playground>
